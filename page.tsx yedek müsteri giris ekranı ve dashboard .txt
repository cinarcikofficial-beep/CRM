'use client';

import { useState, useEffect } from 'react';
import { useRouter } from 'next/navigation';
import { createClient } from '@/utils/supabase/client';
import AddClientForm from './add-client-form';

interface ClientData {
  id: string;
  company_name: string;
  contact_person: string;
  email: string;
  phone: string;
  status: string;
  created_at: string;
}

export default function ClientsPage() {
  const router = useRouter();
  const supabase = createClient();

  const [clients, setClients] = useState<ClientData[]>([]);
  const [searchTerm, setSearchTerm] = useState('');
  const [loading, setLoading] = useState(true);

  // Düzenleme (Inline Edit) State'leri
  const [editingId, setEditingId] = useState<string | null>(null);
  const [editCompanyName, setEditCompanyName] = useState('');
  const [editContactPerson, setEditContactPerson] = useState('');
  const [editEmail, setEditEmail] = useState('');
  const [editPhone, setEditPhone] = useState('');
  const [editStatus, setEditStatus] = useState('');
  const [editLoading, setEditLoading] = useState(false);

  const fetchClients = async () => {
    try {
      setLoading(true);
      const { data, error } = await supabase
        .from('clients')
        .select('*')
        .order('created_at', { ascending: false });

      if (error) throw error;
      setClients(data || []);
    } catch (error: any) {
      alert('Müşteriler yüklenirken hata oluştu: ' + error.message);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchClients();
  }, []);

  const startEditing = (client: ClientData) => {
    setEditingId(client.id);
    setEditCompanyName(client.company_name || '');
    setEditContactPerson(client.contact_person || '');
    setEditEmail(client.email || '');
    setEditPhone(client.phone || '');
    setEditStatus(client.status || 'Potansiyel');
  };

  const handleUpdateClient = async (id: string) => {
    if (!editCompanyName.trim()) return;
    setEditLoading(true);
    try {
      const { error } = await supabase
        .from('clients')
        .update({
          company_name: editCompanyName,
          contact_person: editContactPerson,
          email: editEmail,
          phone: editPhone,
          status: editStatus
        })
        .eq('id', id);

      if (error) throw error;

      setClients((prev) =>
        prev.map((c) =>
          c.id === id
            ? {
                ...c,
                company_name: editCompanyName,
                contact_person: editContactPerson,
                email: editEmail,
                phone: editPhone,
                status: editStatus
              }
            : c
        )
      );
      setEditingId(null);
    } catch (error: any) {
      alert('Güncelleme sırasında hata oluştu: ' + error.message);
    } finally {
      setEditLoading(false);
    }
  };

  const handleDeleteClient = async (id: string) => {
    if (!confirm('Bu müşteriyi silmek istediğinize emin misiniz?')) return;
    try {
      const { error } = await supabase.from('clients').delete().eq('id', id);
      if (error) throw error;
      setClients((prev) => prev.filter((c) => c.id !== id));
    } catch (error: any) {
      alert('Silme işlemi başarısız: ' + error.message);
    }
  };

  const filteredClients = clients.filter((client) =>
    client.company_name?.toLowerCase().includes(searchTerm.toLowerCase()) ||
    client.contact_person?.toLowerCase().includes(searchTerm.toLowerCase())
  );

  return (
    <div className="min-h-screen bg-black text-white p-6">
      <div className="max-w-7xl mx-auto space-y-6">
        
        {/* Üst Form Alanı ve Rapor Sayfası Linki */}
        <div className="bg-zinc-950 border border-zinc-800 p-6 rounded-xl shadow-xl">
          <div className="flex justify-between items-center mb-4">
            <h1 className="text-xl font-black">CRM Müşteri Yönetimi</h1>
            <button
              onClick={() => router.push('/reports')}
              className="text-xs bg-indigo-600 hover:bg-indigo-700 text-white font-bold py-2 px-4 rounded-lg transition-all shadow-md"
            >
              Aktivite Raporu →
            </button>
          </div>
          <AddClientForm onClientAdded={fetchClients} />
        </div>

        {/* Filtreleme Alanı */}
        <div className="flex gap-4">
          <input
            type="text"
            placeholder="Müşteri veya yetkili adı ile filtreleyin..."
            value={searchTerm}
            onChange={(e) => setSearchTerm(e.target.value)}
            className="w-full md:w-1/3 px-4 py-2 bg-zinc-950 border border-zinc-800 rounded-lg text-sm text-white focus:outline-none focus:border-indigo-500 placeholder-zinc-600"
          />
        </div>

        {/* Müşteri Tablo Listesi */}
        <div className="bg-zinc-950 border border-zinc-800 rounded-xl overflow-hidden shadow-xl">
          {loading ? (
            <div className="p-8 text-center text-zinc-500 text-sm">Veriler yükleniyor...</div>
          ) : filteredClients.length === 0 ? (
            <div className="p-8 text-center text-zinc-500 text-sm">Hiç müşteri kaydı bulunamadı.</div>
          ) : (
            <div className="overflow-x-auto">
              <table className="w-full text-left border-collapse text-sm">
                <thead>
                  <tr className="border-b border-zinc-800 bg-zinc-900/50 text-zinc-400 font-medium">
                    <th className="p-4">Müşteri / Firma Adı</th>
                    <th className="p-4">Yetkili Kişi</th>
                    <th className="p-4">İletişim Bilgileri</th>
                    <th className="p-4">Durum</th>
                    <th className="p-4 text-right">İşlemler</th>
                  </tr>
                </thead>
                <tbody>
                  {filteredClients.map((client) => {
                    const isEditing = editingId === client.id;
                    return (
                      <tr key={client.id} className="border-b border-zinc-800 hover:bg-zinc-900/30 transition-colors">
                        
                        {/* Müşteri / Firma Adı */}
                        <td className="p-4 font-bold text-white">
                          {isEditing ? (
                            <input
                              type="text"
                              value={editCompanyName}
                              onChange={(e) => setEditCompanyName(e.target.value)}
                              className="bg-zinc-900 border border-zinc-800 px-2 py-1 rounded text-sm text-white focus:outline-none focus:border-indigo-500"
                            />
                          ) : (
                            client.company_name
                          )}
                        </td>

                        {/* Yetkili Kişi */}
                        <td className="p-4 text-zinc-300">
                          {isEditing ? (
                            <input
                              type="text"
                              value={editContactPerson}
                              onChange={(e) => setEditContactPerson(e.target.value)}
                              className="bg-zinc-900 border border-zinc-800 px-2 py-1 rounded text-sm text-white focus:outline-none focus:border-indigo-500"
                            />
                          ) : (
                            client.contact_person
                          )}
                        </td>

                        {/* İletişim Bilgileri */}
                        <td className="p-4 text-zinc-400 font-mono text-xs">
                          {isEditing ? (
                            <div className="space-y-1">
                              <input
                                type="email"
                                value={editEmail}
                                onChange={(e) => setEditEmail(e.target.value)}
                                className="bg-zinc-900 border border-zinc-800 px-2 py-0.5 rounded text-xs text-white block focus:outline-none"
                                placeholder="E-posta"
                              />
                              <input
                                type="text"
                                value={editPhone}
                                onChange={(e) => setEditPhone(e.target.value)}
                                className="bg-zinc-900 border border-zinc-800 px-2 py-0.5 rounded text-xs text-white block focus:outline-none"
                                placeholder="Telefon"
                              />
                            </div>
                          ) : (
                            <>
                              {client.email} <br /> {client.phone || '-'}
                            </>
                          )}
                        </td>

                        {/* Durum */}
                        <td className="p-4">
                          {isEditing ? (
                            <select
                              value={editStatus}
                              onChange={(e) => setEditStatus(e.target.value)}
                              className="bg-zinc-900 border border-zinc-800 px-2 py-1 rounded text-xs text-white focus:outline-none"
                            >
                              <option value="Potansiyel">Potansiyel</option>
                              <option value="Aktif Müşteri">Aktif Müşteri</option>
                              <option value="Pasif">Pasif</option>
                            </select>
                          ) : (
                            <span className="px-2.5 py-1 rounded-full bg-indigo-500/10 text-indigo-400 text-xs border border-indigo-500/20 font-medium">
                              {client.status || 'Potansiyel'}
                            </span>
                          )}
                        </td>

                        {/* İşlemler */}
                        <td className="p-4 text-right space-x-2 whitespace-nowrap">
                          {isEditing ? (
                            <div className="inline-flex gap-2">
                              <button
                                onClick={() => setEditingId(null)}
                                disabled={editLoading}
                                className="text-xs bg-zinc-900 border border-zinc-800 hover:bg-zinc-800 text-zinc-400 px-2.5 py-1.5 rounded-lg transition-all"
                              >
                                İptal
                              </button>
                              <button
                                onClick={() => handleUpdateClient(client.id)}
                                disabled={editLoading}
                                className="text-xs bg-indigo-600 hover:bg-indigo-700 text-white px-2.5 py-1.5 rounded-lg font-bold transition-all"
                              >
                                {editLoading ? '...' : 'Kaydet'}
                              </button>
                            </div>
                          ) : (
                            <>
                              <button
                                onClick={() => router.push(`/clients/${client.id}`)}
                                className="text-xs bg-zinc-900 border border-zinc-800 hover:bg-zinc-800 text-white px-3 py-1.5 rounded-lg transition-all"
                              >
                                Detay & Notlar →
                              </button>
                              <button
                                onClick={() => startEditing(client)}
                                className="text-zinc-500 hover:text-indigo-400 text-sm p-1.5"
                                title="Müşteriyi Düzenle"
                              >
                                ✏️
                              </button>
                              <button
                                onClick={() => handleDeleteClient(client.id)}
                                className="text-zinc-500 hover:text-red-400 text-sm p-1.5"
                              >
                                🗑️
                              </button>
                            </>
                          )}
                        </td>

                      </tr>
                    );
                  })}
                </tbody>
              </table>
            </div>
          )}
        </div>

      </div>
    </div>
  );
}
